Een geïntegreerd project van Mike van Bree Zes Industriële ICT, Mosa-RT 

                    #include <Adafruit_NeoPixel.h>
                    
                    #define NUMPIXELS 128
                    #define PIN 13
                    
                    Adafruit_NeoPixel neoPixel = Adafruit_NeoPixel(NUMPIXELS, PIN, NEO_RGB + NEO_KHZ800);
                    
                    void setup() {
                      // put your setup code here, to run once:
                      neoPixel.begin();
                      neoPixel.setBrightness(16);
                      neoPixel.show();
                    
                    }
                    
                    void loop() {
                      // put your main code here, to run repeatedly:
                      for (int i = 0; i < NUMPIXELS; i += 2) {
                        neoPixel.setPixelColor(i, neoPixel.Color(255, 0, 255));
                        neoPixel.setPixelColor(i + 1, neoPixel.Color(255, 0, 255));
                        neoPixel.show();
                        delay(200);
                      }
                      delay(150);
                      for (int i = NUMPIXELS; i > -1; i -= 2) {
                        neoPixel.setPixelColor(i, neoPixel.Color(0, 0, 0));
                        neoPixel.setPixelColor(i+1, neoPixel.Color(0, 0, 0));
                        neoPixel.show();
                        delay(200);
                      }
                      delay(150);
                    }
